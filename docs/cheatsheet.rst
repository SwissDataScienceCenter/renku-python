    \section{Configuration Options}
    \begin{itemize}
    \item
    \textbf{registry}: Docker image registry
    \item
    \textbf{zenodo.access\_token}: Token for Zenodo export
    \item
    \textbf{dataverse.access\_token}: Token for Dataverse export
    \item
    \textbf{show\_lfs\_message}: Whether to show the LFS warning message
    \item
    \textbf{lfs\_threshold}: Size threshold below which files aren't added to LFS
    \end{itemize}
